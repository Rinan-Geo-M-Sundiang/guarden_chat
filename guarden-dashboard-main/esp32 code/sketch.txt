// =============================================================================
// main.ino — Guarden ESP32
// Reads 3× DHT22 + soil sensors, uploads to Firebase, and applies thresholds
// set from the dashboard in real time.
// =============================================================================

#include <DHT.h>
#include "firebase_uploader.h"

// ─── Pin definitions ──────────────────────────────────────────────────────────
#define DHT_PIN_1   45
#define DHT_PIN_2   46
#define DHT_PIN_3   47
#define SOIL_PIN_1  1
#define SOIL_PIN_2  2
#define SOIL_PIN_3  10
#define RELAY_PIN   48

// ─── Soil sensor calibration ──────────────────────────────────────────────────
#define SOIL_DRY  3200
#define SOIL_WET   800

#define DHT_TYPE DHT22
DHT dht1(DHT_PIN_1, DHT_TYPE);
DHT dht2(DHT_PIN_2, DHT_TYPE);
DHT dht3(DHT_PIN_3, DHT_TYPE);

// ─── Timing (all in ms) ───────────────────────────────────────────────────────
#define SENSOR_INTERVAL     5000UL
#define UPLOAD_INTERVAL    10000UL
#define THRESHOLD_INTERVAL  3000UL
#define RELAY_INTERVAL      3000UL

// ─── Runtime state ────────────────────────────────────────────────────────────
unsigned long lastSensorRead       = 0;
unsigned long lastUpload           = 0;
unsigned long lastThresholdCheck   = 0;
unsigned long lastRelayCheck       = 0;
unsigned long lastKnownThresholdTs = 0;

float temps[NUM_ZONES]      = {};
float hums[NUM_ZONES]       = {};
float moistures[NUM_ZONES]  = {};
bool  validZones[NUM_ZONES] = {};

bool autoRelayState       = false;
bool remoteOverride       = false;
bool lastLoggedRelayState = false;
bool lastLoggedOverride   = false;


// ─── Helpers ──────────────────────────────────────────────────────────────────

float soilToPercent(int raw) {
    float pct = 100.0f * float(SOIL_DRY - raw) / float(SOIL_DRY - SOIL_WET);
    return constrain(pct, 0.0f, 100.0f);
}

void readSensors() {
    DHT* dhts[NUM_ZONES]     = { &dht1, &dht2, &dht3 };
    int  soilPins[NUM_ZONES] = { SOIL_PIN_1, SOIL_PIN_2, SOIL_PIN_3 };

    for (int i = 0; i < NUM_ZONES; i++) {
        float t = dhts[i]->readTemperature();
        float h = dhts[i]->readHumidity();

        if (isnan(t) || isnan(h)) {
            validZones[i] = false;
            Serial.printf("DHT_ERR zone-%d\n", i + 1);
            continue;
        }
        temps[i]      = t;
        hums[i]       = h;
        moistures[i]  = soilToPercent(analogRead(soilPins[i]));
        validZones[i] = true;
        Serial.printf("ZONE-%d  temp=%.1f  hum=%.1f  moist=%.1f\n",
                      i + 1, t, h, moistures[i]);
    }
}

float avgOf(float arr[], bool valid[]) {
    float sum = 0; int n = 0;
    for (int i = 0; i < NUM_ZONES; i++) {
        if (valid[i]) { sum += arr[i]; n++; }
    }
    return n ? sum / n : 0;
}

// Returns true if at least one zone has a valid reading
bool anyZoneValid() {
    for (int i = 0; i < NUM_ZONES; i++) {
        if (validZones[i]) return true;
    }
    return false;
}

// ─── NTP wait helper ──────────────────────────────────────────────────────────
bool waitForNTP(unsigned long timeoutMs = 10000UL) {
    Serial.print("NTP syncing");
    unsigned long t0 = millis();
    while (millis() - t0 < timeoutMs) {
        time_t now;
        time(&now);
        if (now > 1000000000UL) {
            Serial.printf(" OK  epoch=%lu\n", (unsigned long)now);
            return true;
        }
        delay(500);
        Serial.print('.');
    }
    Serial.println(" TIMEOUT — uploads will be skipped until sync");
    return false;
}

// ─── Threshold polling ────────────────────────────────────────────────────────
void pollThresholdsIfNeeded() {
    if (millis() - lastThresholdCheck < THRESHOLD_INTERVAL) return;
    lastThresholdCheck = millis();
    if (WiFi.status() != WL_CONNECTED) return;

    unsigned long remoteTs = fetchThresholdUpdateTime();
    if (remoteTs == 0) return;

    if (remoteTs != lastKnownThresholdTs) {
        Serial.printf("THRESH_CHANGE  remote=%lu  local=%lu — fetching...\n",
                      remoteTs, lastKnownThresholdTs);
        updateRemoteThresholds();
        lastKnownThresholdTs = remoteTs;

        Thresholds t = getRemoteThresholds();
        Serial.printf("THRESH_APPLIED  TempOn=%.1f  TempOff=%.1f  "
                      "MoistWarn=%.0f  TempWarn=%.1f  HumWarn=%.0f\n",
                      t.tempOn, t.tempOff,
                      t.moistureWarn, t.temperatureWarn, t.humidityWarn);
    }
}

// ─── Relay control ────────────────────────────────────────────────────────────
void updateRelay(float avgTemp) {
    if (millis() - lastRelayCheck >= RELAY_INTERVAL) {
        lastRelayCheck = millis();
        remoteOverride = fetchRemoteRelayState();
    }

    if (remoteOverride) {
        digitalWrite(RELAY_PIN, HIGH);
        if (!lastLoggedOverride) {
            Serial.println("RELAY ON  (dashboard override)");
            lastLoggedOverride   = true;
            lastLoggedRelayState = true;
        }
        return;
    }

    lastLoggedOverride = false;

    Thresholds t = getRemoteThresholds();

    if (!autoRelayState && avgTemp >= t.tempOn)  autoRelayState = true;
    if ( autoRelayState && avgTemp <= t.tempOff) autoRelayState = false;

    digitalWrite(RELAY_PIN, autoRelayState ? HIGH : LOW);

    if (autoRelayState != lastLoggedRelayState) {
        Serial.printf("RELAY %s  (auto  avgTemp=%.1f  on>=%.1f  off<=%.1f)\n",
                      autoRelayState ? "ON" : "OFF",
                      avgTemp, t.tempOn, t.tempOff);
        lastLoggedRelayState = autoRelayState;
    }
}


// =============================================================================
// setup()
// =============================================================================
void setup() {
    Serial.begin(115200);
    delay(300);
    Serial.println("\n=== Guarden booting ===");

    pinMode(RELAY_PIN, OUTPUT);
    digitalWrite(RELAY_PIN, LOW);

    dht1.begin(); dht2.begin(); dht3.begin();

    // ── DHT22 requires at least 2 s after power-on before first read ──────────
    Serial.println("Waiting for DHT22 sensors to settle...");
    delay(2500);

    // ── Initial sensor read in setup() ────────────────────────────────────────
    Serial.println("Initial sensor read...");
    readSensors();

    uploaderBegin();   // starts WiFi

    // ── Wait for WiFi ─────────────────────────────────────────────────────────
    Serial.print("WiFi connecting");
    unsigned long t0 = millis();
    while (WiFi.status() != WL_CONNECTED && millis() - t0 < 15000UL) {
        delay(500); Serial.print('.');
    }
    Serial.println();

    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("WiFi connected");

        configTime(NTP_GMT_OFFSET, NTP_DST_OFFSET, NTP_SERVER);
        waitForNTP(10000UL);

        updateRemoteThresholds();
        lastKnownThresholdTs = fetchThresholdUpdateTime();

        Thresholds t = getRemoteThresholds();
        Serial.printf("BOOT THRESHOLDS  TempOn=%.1f  TempOff=%.1f  "
                      "MoistWarn=%.0f  TempWarn=%.1f  HumWarn=%.0f\n",
                      t.tempOn, t.tempOff,
                      t.moistureWarn, t.temperatureWarn, t.humidityWarn);
    } else {
        Serial.println("WiFi timeout — using default thresholds, NTP unavailable");
    }

    lastSensorRead     = millis();
    lastUpload         = millis();
    lastThresholdCheck = millis();
    lastRelayCheck     = millis();
}


// =============================================================================
// loop()
// =============================================================================
void loop() {

    // 1. Check for threshold changes from the dashboard (lightweight)
    pollThresholdsIfNeeded();

    // 2. Read sensors on interval
    if (millis() - lastSensorRead >= SENSOR_INTERVAL) {
        lastSensorRead = millis();
        readSensors();
    }

    // 3. Drive relay using live dashboard thresholds
    updateRelay(avgOf(temps, validZones));

    // 4. Push to Firebase
    if (millis() - lastUpload >= UPLOAD_INTERVAL) {
        lastUpload = millis();

        if (!anyZoneValid()) {
            Serial.println("UPLOAD_SKIP: no valid sensor readings this cycle");
            return;
        }

        float avgTemp     = avgOf(temps,     validZones);
        float avgHum      = avgOf(hums,      validZones);
        float avgMoisture = avgOf(moistures, validZones);  // FIX: was missing

        Serial.printf("AVG  temp=%.1f  hum=%.1f  moisture=%.1f\n",
                      avgTemp, avgHum, avgMoisture);

        uploadOrBuffer(avgTemp, avgHum, avgMoisture, temps, hums, moistures, validZones);
    }

    delay(100);
}
