// =============================================================================
// firebase_uploader.h — WiFi + NTP + Firebase RTDB via direct REST (no SDK)
//
// REST endpoints:
//   PUT /sensors/zone-1.json?auth=SECRET  → DHT22 #1 snapshot
//   PUT /sensors/zone-2.json?auth=SECRET  → DHT22 #2 snapshot
//   PUT /sensors/zone-3.json?auth=SECRET  → DHT22 #3 snapshot
//   PUT /averages/temperature.json?auth=SECRET
//   PUT /averages/humidity.json?auth=SECRET
//
// Each zone write satisfies DB rule: hasChildren(['moisture','temperature','humidity','updatedAt'])
// ⚠ WARNING: DB secret is embedded. Do NOT push to public repos.
// =============================================================================

// ─── NTP ──────────────────────────────────────────────────────────────────────

#define NTP_GMT_OFFSET 28800L   // UTC+8 (Philippines)
#define NTP_DST_OFFSET 0L

// ─── Zone / sensor count ──────────────────────────────────────────────────────
#define NUM_ZONES 3   // One DHT22 per zone: zone-1, zone-2, zone-3

// ─── Threshold Structure ──────────────────────────────────────────────────────
struct Thresholds {
    float tempOn;
    float tempOff;
    float moistureWarn;
    float temperatureWarn;
    float humidityWarn;
};

static OfflineBuffer _offBuf;
static bool          _ntpSynced = false;

static void _initNTP() {
    if (_ntpSynced) return;
    configTime(NTP_GMT_OFFSET, NTP_DST_OFFSET, NTP_SERVER);
    _ntpSynced = true;
    Serial.println(F("NTP: sync requested"));
}

static unsigned long _epoch() {
    time_t now;
    time(&now);
    return (now > 1000000000UL) ? (unsigned long)now : 0UL;
}

// ── Single HTTP PUT — returns true on any 2xx ─────────────────────────────────
static bool _httpPut(WiFiClientSecure& client, const char* url, const char* body) {
    HTTPClient http;
    if (!http.begin(client, url)) {
        Serial.print(F("FB_ERR: http.begin failed → "));
        Serial.println(url);
        return false;
    }
    http.addHeader(F("Content-Type"), F("application/json"));
    int code = http.PUT((uint8_t*)body, strlen(body));
    http.end();

    if (code < 200 || code > 299) {
        Serial.print(F("FB_ERR HTTP "));
        Serial.print(code);
        Serial.print(F(" → "));
        Serial.println(url);
        return false;
    }
    return true;
}

// ── Push one zone snapshot to /sensors/zone-{N} ───────────────────────────────
// Writes all four fields required by DB validation rule:
//   { temperature, humidity, moisture, updatedAt }
// Only the values of the single DHT22 assigned to this zone are written.
static bool _pushZoneSensor(WiFiClientSecure& client,
                            int           zone,     // 1-based
                            float         temp,     // from DHT22 #zone
                            float         hum,      // from DHT22 #zone
                            float         moisture, // 0.0 if no soil sensor
                            unsigned long ts)
{
    // Clamp moisture to DB-validated range [0, 100]
    if (moisture < 0.0f)   moisture = 0.0f;
    if (moisture > 100.0f) moisture = 100.0f;

    char body[128];
    snprintf(body, sizeof(body),
             "{\"temperature\":%.2f,\"humidity\":%.2f,\"moisture\":%.2f,\"updatedAt\":%lu}",
             temp, hum, moisture, (unsigned long long)ts * 1000ULL);

    // Targets:
    //   zone=1 → /sensors/zone-1.json
    //   zone=2 → /sensors/zone-2.json
    //   zone=3 → /sensors/zone-3.json
    char url[192];
    snprintf(url, sizeof(url),
             "%s/sensors/zone-%d.json?auth=%s",
             FIREBASE_DB_URL, zone, FIREBASE_SECRET);

    return _httpPut(client, url, body);
}

// ── Push averages to /averages/temperature and /averages/humidity ─────────────
static bool _pushAverages(WiFiClientSecure& client,
                          float         avgTemp,
                          float         avgHum,
                          unsigned long /*ts*/)   // ts unused for plain numeric nodes
{
    char valBuf[32];
    char url[192];
    bool ok = true;

    // → /averages/temperature
    snprintf(valBuf, sizeof(valBuf), "%.2f", avgTemp);
    snprintf(url, sizeof(url),
             "%s/averages/temperature.json?auth=%s",
             FIREBASE_DB_URL, FIREBASE_SECRET);
    if (!_httpPut(client, url, valBuf)) ok = false;

    // → /averages/humidity
    snprintf(valBuf, sizeof(valBuf), "%.2f", avgHum);
    snprintf(url, sizeof(url),
             "%s/averages/humidity.json?auth=%s",
             FIREBASE_DB_URL, FIREBASE_SECRET);
    if (!_httpPut(client, url, valBuf)) ok = false;

    return ok;
}

// ── Full upload cycle ─────────────────────────────────────────────────────────
// Per-zone: only zones where validZones[i]==true are written.
//   DHT22 #1 (i=0) → zone-1   (its own temp/hum only)
//   DHT22 #2 (i=1) → zone-2   (its own temp/hum only)
//   DHT22 #3 (i=2) → zone-3   (its own temp/hum only)
// Failed zones are skipped entirely — no stale data ever written.
static bool _pushAll(float  avgTemp,
                     float  avgHum,
                     float  temps[NUM_ZONES],
                     float  hums[NUM_ZONES],
                     float  moistures[NUM_ZONES],
                     bool   validZones[NUM_ZONES],
                     unsigned long ts)
{
    WiFiClientSecure client;
    client.setInsecure();   // Skip cert verify (acceptable for Wokwi sim)

    bool ok = true;

    // ── Per-zone writes (only the single sensor assigned to each zone) ─────
    for (int i = 0; i < NUM_ZONES; i++) {
        int zone = i + 1;   // i=0→zone-1, i=1→zone-2, i=2→zone-3

        if (!validZones[i]) {
            Serial.print(F("ZONE_SKIP: zone-"));
            Serial.print(zone);
            Serial.println(F(" (sensor error)"));
            continue;
        }

        if (!_pushZoneSensor(client, zone, temps[i], hums[i], moistures[i], ts)) {
            Serial.print(F("ZONE_ERR: zone-"));
            Serial.println(zone);
            ok = false;
        } else {
            Serial.printf("ZONE_OK zone-%d: temp=%.2f hum=%.2f moist=%.2f\n",
                          zone, temps[i], hums[i], moistures[i]);
        }
    }

    // ── Average writes ─────────────────────────────────────────────────────
    if (!_pushAverages(client, avgTemp, avgHum, ts)) {
        Serial.println(F("AVG_ERR: failed to write averages"));
        ok = false;
    } else {
        Serial.printf("AVG_OK: temp=%.2f hum=%.2f ts=%lu\n", avgTemp, avgHum, ts);
    }

    return ok;
}

void uploaderBegin() {
    _offBuf.begin();
    WiFi.begin(WIFI_SSID, WIFI_PASS, WIFI_CHANNEL);
    Serial.println(F("WIFI: connecting..."));
}

// ---------------------------------------------------------------------------
// uploadOrBuffer — called once per sketch loop() cycle
//
//   avgTemp      — average across all valid zones
//   avgHum       — average across all valid zones
//   temps[]      — per-zone temperature  [NUM_ZONES]  (index 0 = DHT22 #1)
//   hums[]       — per-zone humidity     [NUM_ZONES]  (index 0 = DHT22 #1)
//   moistures[]  — per-zone moisture%    [NUM_ZONES]  (pass zeros if none)
//   validZones[] — true only if that DHT22 read succeeded this cycle
// ---------------------------------------------------------------------------
void uploadOrBuffer(float avgTemp,
                    float avgHum,
                    float temps[NUM_ZONES],
                    float hums[NUM_ZONES],
                    float moistures[NUM_ZONES],
                    bool  validZones[NUM_ZONES])
{
    // ── WiFi offline → buffer averages, skip zone push ────────────────────
    if (WiFi.status() != WL_CONNECTED) {
        _offBuf.push(avgTemp, avgHum, _epoch());
        Serial.print(F("OFFLINE: buffered (queue="));
        Serial.print(_offBuf.count());
        Serial.println(')');
        return;
    }

    _initNTP();

    // ── Drain buffered averages from previous offline periods ─────────────
    if (_offBuf.count() > 0) {
        Serial.print(F("SYNC: draining "));
        Serial.print(_offBuf.count());
        Serial.println(F(" buffered average(s)"));

        WiFiClientSecure client;
        client.setInsecure();

        Reading r;
        while (_offBuf.pop(r)) {
            if (!_pushAverages(client, r.temp, r.hum, r.ts)) {
                _offBuf.push(r.temp, r.hum, r.ts);
                Serial.println(F("SYNC: REST error, aborting drain"));
                break;
            }
            delay(60);
        }
    }

    // ── Live upload ───────────────────────────────────────────────────────
    unsigned long ts = _epoch();
    if (_pushAll(avgTemp, avgHum, temps, hums, moistures, validZones, ts)) {
        Serial.printf("UPLOAD_OK ts=%lu\n", ts);
    } else {
        _offBuf.push(avgTemp, avgHum, ts);
        Serial.println(F("UPLOAD_FAIL: averages buffered"));
    }
}

bool fetchRemoteRelayState() {
    if (WiFi.status() != WL_CONNECTED) return false;
    
    WiFiClientSecure client;
    client.setInsecure();
    HTTPClient http;
    
    char url[192];
    snprintf(url, sizeof(url), "%s/control/relay.json?auth=%s", FIREBASE_DB_URL, FIREBASE_SECRET);
    
    if (http.begin(client, url)) {
        int code = http.GET();
        if (code == 200) {
            String payload = http.getString();
            http.end();
            return (payload == "true"); // Returns true if Firebase says "true"
        }
    }
    http.end();
    return false;
}

// ─── THRESHOLD MANAGEMENT ──────────────────────────────────────────────────
// Fetch the "lastUpdated" timestamp to detect if thresholds changed
unsigned long fetchThresholdUpdateTime() {
    if (WiFi.status() != WL_CONNECTED) {
        Serial.println("TIMESTAMP_CHECK: WiFi not connected");
        return 0;
    }

    WiFiClientSecure client;
    client.setInsecure();
    HTTPClient http;

    char url[192];
    snprintf(url, sizeof(url), "%s/config/thresholds_updated.json?auth=%s", FIREBASE_DB_URL, FIREBASE_SECRET);

    Serial.printf("TIMESTAMP_CHECK: Fetching from %s\n", url);

    if (!http.begin(client, url)) {
        Serial.println("TIMESTAMP_CHECK: http.begin failed");
        return 0;
    }

    int code = http.GET();
    String payload = http.getString();
    http.end();

    Serial.printf("TIMESTAMP_CHECK: HTTP %d, payload=%s\n", code, payload.c_str());

    if (code != 200) {
        return 0;
    }

    // Firebase returns the timestamp as a number, e.g., "1714734600000"
    unsigned long timestamp = payload.toInt();
    Serial.printf("TIMESTAMP_CHECK: Parsed timestamp=%lu\n", timestamp);
    return timestamp;
}
Thresholds fetchRemoteThresholds() {
    Thresholds defaults = {40.0f, 15.0f, 40.0f, 20.0f, 50.0f};

    if (WiFi.status() != WL_CONNECTED) return defaults;

    WiFiClientSecure client;
    client.setInsecure();
    HTTPClient http;

    char url[192];
    snprintf(url, sizeof(url), "%s/config/thresholds.json?auth=%s", FIREBASE_DB_URL, FIREBASE_SECRET);

    if (!http.begin(client, url)) return defaults;

    int code = http.GET();
    if (code != 200) {
        http.end();
        return defaults;
    }

    String payload = http.getString();
    http.end();

    // Simple parsing - Firebase REST returns: {"tempOn":40,"tempOff":15,"moistureWarn":40,"temperatureWarn":20,"humidityWarn":50}
    Thresholds result = defaults;
    Serial.printf("RAW_JSON: %s\n", payload.c_str());

    // Helper to extract float value after a key
    auto extractFloat = [](const String& json, const char* key, float def) -> float {
        int pos = json.indexOf(key);
        if (pos < 0) return def;
        pos = json.indexOf(":", pos);
        if (pos < 0) return def;
        pos++; // skip the colon
        // Skip whitespace
        while (pos < json.length() && (json[pos] == ' ' || json[pos] == '\t')) pos++;

        int endPos = pos;
        while (endPos < json.length() && json[endPos] != ',' && json[endPos] != '}') endPos++;

        String numStr = json.substring(pos, endPos);
        numStr.trim();
        float val = numStr.toFloat();
        return (val > 0 || numStr == "0") ? val : def;  // Return def if parsing failed
    };

    result.tempOn = extractFloat(payload, "tempOn", defaults.tempOn);
    result.tempOff = extractFloat(payload, "tempOff", defaults.tempOff);
    result.moistureWarn = extractFloat(payload, "moistureWarn", defaults.moistureWarn);
    result.temperatureWarn = extractFloat(payload, "temperatureWarn", defaults.temperatureWarn);
    result.humidityWarn = extractFloat(payload, "humidityWarn", defaults.humidityWarn);

    Serial.printf("THRESH_PARSED: TempOn=%.1f TempOff=%.1f MoistWarn=%.0f TempWarn=%.1f HumWarn=%.0f\n",
                  result.tempOn, result.tempOff, result.moistureWarn, result.temperatureWarn, result.humidityWarn);

    return result;
}

// Global cache for remote thresholds (persists between calls)
static Thresholds _remoteThresholds = {40.0f, 15.0f, 40.0f, 20.0f, 50.0f};

// Get cached thresholds (fast, no Firebase call)
Thresholds getRemoteThresholds() {
    return _remoteThresholds;
}

// Update cache by fetching fresh thresholds from Firebase
void updateRemoteThresholds() {
    _remoteThresholds = fetchRemoteThresholds();
}
